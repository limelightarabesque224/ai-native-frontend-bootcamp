### 4. 数据库 Schema

```prisma
model Product {
  id          String   @id @default(cuid())
  name        String
  description String
  price       Float
  stock       Int
  category    String
  imageUrl    String?
  createdAt   DateTime @default(now())
  updatedAt   DateTime @updatedAt
  orders      OrderItem[]
}

model Order {
  id         String      @id @default(cuid())
  userId     String
  status     OrderStatus
  total      Float
  items      OrderItem[]
  createdAt  DateTime    @default(now())
  updatedAt  DateTime    @updatedAt
}

model OrderItem {
  id        String  @id @default(cuid())
  orderId   String
  productId String
  quantity  Int
  price     Float
  order     Order   @relation(fields: [orderId], references: [id])
  product   Product @relation(fields: [productId], references: [id])
}

enum OrderStatus {
  PENDING
  PROCESSING
  SHIPPED
  DELIVERED
  CANCELLED
}
```

### 5. AI 功能实现

#### 5.1 AI 生成商品描述

```ts
// packages/ai/src/generate-description.ts
import { generateText } from 'ai'
import { openai } from '@ai-sdk/openai'

export async function generateProductDescription(
  name: string,
  category: string,
  features: string[]
) {
  const { text } = await generateText({
    model: openai('gpt-4o'),
    prompt: `为以下商品生成吸引人的描述：
商品名称：${name}
类别：${category}
特点：${features.join(', ')}

要求：
- 100-150 字
- 突出卖点
- 适合电商平台`,
  })
  return text
}
```

#### 5.2 AI 客服工具调用

```ts
// packages/ai/src/customer-service.ts
import { streamText } from 'ai'
import { openai } from '@ai-sdk/openai'
import { z } from 'zod'

const tools = {
  queryOrder: {
    description: '查询订单信息',
    parameters: z.object({
      orderId: z.string(),
    }),
    execute: async ({ orderId }) => {
      const order = await db.order.findUnique({
        where: { id: orderId },
        include: { items: true },
      })
      return order
    },
  },
  queryProduct: {
    description: '查询商品信息',
    parameters: z.object({
      productName: z.string(),
    }),
    execute: async ({ productName }) => {
      const products = await db.product.findMany({
        where: {
          name: { contains: productName },
        },
      })
      return products
    },
  },
}

export async function chatWithAI(messages: Message[]) {
  const result = streamText({
    model: openai('gpt-4o'),
    messages,
    tools,
    system: '你是一个电商客服助手，可以帮助用户查询订单和商品信息。',
  })
  return result.toDataStreamResponse()
}
```
